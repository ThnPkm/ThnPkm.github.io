from PIL import Image

image = Image.open("(x,y,bin).gif")
new_image = Image.new("RGB",(39,39),(255, 255, 255)) #背景为宽高39的白色

my_list = [] #存每帧的rgb

try:
    while True:
        image.seek(image.tell() + 1)  # 定位gif的帧从1到1089
        my_list.append(image.getpixel((0, 0)))  # 获取每帧的rgb值
        # print(my_list)
except EOFError:
    pass

for x,y,bin in my_list:
    if bin == 1:
        new_image.putpixel((x,y),(0,0,0)) #如果bin为1,则这块像素为黑
new_image.save("flag.png")
